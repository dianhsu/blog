    bool areOccurrencesEqual(string s) {
        vector<int> vec(26, 0);
        for(auto c: s){
            vec[c - 'a']++;
        }
        int tmp = -1;
        for(int i = 0; i < 26; ++i){
            if(vec[i] != 0){
                if(tmp == -1){
                    tmp = vec[i];
                }else if(tmp != vec[i]){
                    return false;
                }
            }
        }
        return true;
    }
};
